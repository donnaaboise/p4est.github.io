134 /*@unused@*/
135 static inline int
136 p4est_lnodes_decode (p4est_lnodes_code_t face_code, int hanging_face[4])
137 {
138  P4EST_ASSERT (face_code >= 0);
139 
140  if (face_code) {
141  int i;
142  int8_t c = face_code & 0x03;
143  int f;
144  int8_t work = face_code >> 2;
145 
146  memset (hanging_face, -1, 4 * sizeof (int));
147 
148  for (i = 0; i < 2; ++i) {
149  f = p4est_corner_faces[c][i];
150  hanging_face[f] = (work & 0x01) ? p4est_corner_face_corners[c][f] : -1;
151  work >>= 1;
152  }
153 
154  return 1;
155  }
156  else {
157  return 0;
158  }
159 }
